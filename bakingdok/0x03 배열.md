#include <bits/stdc++.h>
using namespace std;

void insert(int idx, int num, int arr[], int& len){
    if(idx == len) {
        arr[idx] = num;
    } else {
        for(int i=len; i>idx; i--) {
            arr[i] = arr[i-1];
        }
        arr[idx] = num;
    }
    len++;
}

void erase(int idx, int arr[], int& len){
  if(idx == (len-1)) {
      len--;
  } else {
      for(int i=idx; i<len; i++) {
          arr[i] = arr[i+1];
      }
      len--;
  }
}

void printArr(int arr[], int& len){
  for(int i = 0; i < len; i++) cout << arr[i] << ' ';
  cout << "\n\n";
}

### 틀렸던 이유

- 매 회차마다 배열을 초기화해야하는데 초기화하지 않고 진행하여 전의 결과와 섞이는 문제 발생

### 좀 더 빨리 실행하는 코드

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(void) {
    string first, second;
    int n;
    cin >> n;

    while(n>0) {
        int arr[26] = { };
        cin>>first>>second;

        for(auto c:first) arr[c-'a']++;
        for(auto c:second) arr[c-'a']--;

        bool isPossible = true;
        for(int a:arr) {
            if(a != 0) isPossible = false;
        }
        if (isPossible) {
            cout << "Possible" << '\n';
        } else {
            cout << "Impossible" << '\n';
        }
        n--;
    }
}
```

- equal 함수를 사용하지 않고 동일한 배열에서 값을 빼는 방식으로 진행
